sampler2D implicitInput : register(s0);
float4 basecolor : register(c0);
float4 highoctavescolor : register(c1);
float4 lowoctavescolor : register(c2);

//--------------------------------------------------------------------------------------
// Pixel Shader
//--------------------------------------------------------------------------------------
float4 main(float2 uv : TEXCOORD) : COLOR
{
    float octave = uv.x;
    float4 octaveColor = lerp(highoctavescolor, lowoctavescolor, octave);
    return lerp(basecolor, octaveColor, 1 - uv.y);
}
